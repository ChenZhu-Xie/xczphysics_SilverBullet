
## Go to today botton

1. [sleek interactive floating journal calendar](https://community.silverbullet.md/t/sleek-interactive-floating-journal-calendar/3680/41?u=chenzhu-xie) #community #silverbullet

```space-lua
actionButton.define {
  icon = "sun",
  description = "Go to Today",
  run = function()
    local date = os.date("*t")
    local monthNames = {"January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"}
    local year = tostring(date.year)
    local month = string.format("%02d", date.month)
    local day = string.format("%02d", date.day)
    local monthname = monthNames[date.month]
    local path = "Journal/" .. year .. "/" .. month .. "-" .. monthname .. "/" .. year .. "-" .. month .. "-" .. day
    editor.navigate(path)
  end
}
```
